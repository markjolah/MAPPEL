70 {
71  std::type_index pos_dist = prior.component_types()[dim];
72  std::type_index beta_dist(typeid(prior_hessian::ScaledSymmetricBetaDist));
73  std::type_index normal_dist(typeid(prior_hessian::TruncatedNormalDist));
74  auto hyperparams = prior.params();
75  if(pos_dist == beta_dist){
76  double beta_pos = hyperparams(dim);
77  double mean_I = hyperparams(2);
78  double kappa_I = hyperparams(3);
79  double mean_bg = size((dim+1)%2) * hyperparams(4); //Amplify bg to account for summation
80  double kappa_bg = hyperparams(5);
81  double alpha_sigma = hyperparams(6);
82  return {size(dim),Gauss1DSumModelT::make_prior_beta_position(
83  size(dim), beta_pos, mean_I, kappa_I, mean_bg, kappa_bg, min_sigma(dim), max_sigma(dim),alpha_sigma)};
84  } else if(pos_dist == normal_dist) {
85  double sigma_pos = hyperparams(2*dim+1);
86  double mean_I = hyperparams(4);
87  double kappa_I = hyperparams(5);
88  double mean_bg = size((dim+1)%2) * hyperparams(6); //Amplify bg to account for summation
89  double kappa_bg = hyperparams(7);

91  return {size(dim),Gauss1DSumModelT::make_prior_normal_position(

358 void Gauss2DsModel::pixel_hess_update(int i, int j, const Stencil &s, double dm_ratio_m1, double dmm_ratio,
359  ParamT &grad, MatT &hess) const
360 {
361  auto pgrad = make_param();
362  pixel_grad(i,j,s,pgrad);
363  double I = s.I();
364  /* Update grad */
365  grad += dm_ratio_m1*pgrad;
366  //Update Hessian
367  //On Diagonal
368  hess(0,0) += dm_ratio_m1 * I/s.sigmaX() * s.DXS(i) * s.Y(j); //xx
369  hess(1,1) += dm_ratio_m1 * I/s.sigmaY() * s.DYS(j) * s.X(i); //yy
370  hess(4,4) += dm_ratio_m1 * I * (s.X(i)*s.DYS2(j) + 2.*s.DXS(i)*s.DYS(j) + s.Y(j)*s.DXS2(i)); //SS
371  //Off Diagonal
372  hess(0,1) += dm_ratio_m1 * I * s.DX(i) * s.DY(j); //xy
373  hess(0,4) += dm_ratio_m1 * I * (s.Y(j)*s.DXSX(i) + s.DX(i)*s.DYS(j)); //xS
374  hess(1,4) += dm_ratio_m1 * I * (s.X(i)*s.DYSY(j) + s.DY(j)*s.DXS(i)); //yS
375  //Off Diagonal with respect to I
376  hess(0,2) += dm_ratio_m1 * pgrad(0) / I; //xI
377  hess(1,2) += dm_ratio_m1 * pgrad(1) / I; //yI
378  hess(2,4) += dm_ratio_m1 * pgrad(4) / I; //IS
379  //This is the pixel-gradient dependenent part of the hessian
380  for(IdxT c=0; c<hess.n_cols; c++) for(IdxT r=0; r<=c; r++)
381  hess(r,c) -= dmm_ratio * pgrad(r) * pgrad(c);
382 }

384 Gauss2DsModel::Stencil
385 Gauss2DsModel::initial_theta_estimate(const ImageT &im, const ParamT &theta_init,
386  const std::string &estimator_method)
387 {
388  double x_pos = 0;
389  double y_pos = 0;
390  double I = 0;
391  double bg = 0;
392  double sigma_ratio = 0;
393  if(theta_init.n_elem == num_params){
394  if(theta_in_bounds(theta_init)) return make_stencil(theta_init);
395  x_pos = theta_init(0);
396  y_pos = theta_init(1);
397  I = theta_init(2);
398  bg = theta_init(3);
399  sigma_ratio = theta_init(4);
400  }
401  Gauss2DsModel::ImageT x_im = arma::sum(im,0).t();
402  Gauss2DsModel::ImageT y_im = arma::sum(im,1);
403  auto x_est = methods::estimate_max(x_model,x_im,estimator_method);
404  auto y_est = methods::estimate_max(y_model,y_im,estimator_method);
405 
406  if(x_pos <= lbound(0) || x_pos >= ubound(0) || !std::isfinite(x_pos)) x_pos = x_est.theta(0);
407  if(y_pos <= lbound(1) || y_pos >= ubound(1) || !std::isfinite(y_pos)) y_pos = y_est.theta(0);
408  if(I <= lbound(2) || I >= ubound(2) || !std::isfinite(I)) {
409  //max of X and Y est of I
410  I = std::max(x_est.theta(1), y_est.theta(1));
411  }
412  if(bg <= lbound(3) || bg >= ubound(3) || !std::isfinite(bg)) {
413  //mean of X and Y est of bg corrected for 1D vs 2D interpretation of bg
414  bg = .5*(x_est.theta(2)/size(1) + y_est.theta(2)/size(0));
415  }
416  if(sigma_ratio <= lbound(4) || sigma_ratio >= ubound(4) || !std::isfinite(sigma_ratio)) {
417  //mean of X and Y sigma_ratio. 1D models report sigma in pixels, convert to ratio to min_sigma.
418  sigma_ratio = .5*(x_est.theta(3)/min_sigma(0) + y_est.theta(3)/min_sigma(1));
419  }
420  return make_stencil(ParamT{x_pos, y_pos, I, bg, sigma_ratio});
421 }
