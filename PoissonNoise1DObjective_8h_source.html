162  ReturnIfSubclassT<void,Model,PoissonNoise1DObjective>
163  grad2(const Model &model, const ModelDataT<Model> &im, const StencilT<Model> &s,
164  ParamT<Model> &grad_val, ParamT<Model> &grad2_val)
165  {
166  auto pixel_grad = model.make_param();
167  auto pixel_grad2 = model.make_param();
168  grad_val.zeros();
169  grad2_val.zeros();
170  for(ImageCoordT<Model> i=0; i<model.get_size(); i++){
171  if(!std::isfinite(im(i))) continue; /* Skip non-finite image values as they are assumed masked */
172  /* Compute model value and ratios */
173  double model_val = model.pixel_model_value(i,s);
174  double dm_ratio = im(i)/model_val;
175  double dm_ratio_m1 = dm_ratio-1;
176  double dmm_ratio = dm_ratio/model_val;
177  model.pixel_grad(i,s,pixel_grad);
178  model.pixel_grad2(i,s,pixel_grad2);
179  grad_val += dm_ratio_m1*pixel_grad;
180  grad2_val += dm_ratio_m1*pixel_grad2 - dmm_ratio*pixel_grad%pixel_grad;
181  }
