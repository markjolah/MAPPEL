192 void fill_DXS2_stencil(int size, double stencil[], const double dx[], const double Gx[], const double DXS[], double theta_sigma)
193 {
194  double sigma2=theta_sigma*theta_sigma;
195  double norm=1./sqrt2pi/(sigma2*sigma2*theta_sigma); /* 1/(sqrt(2*pi)*sigma^5) */
196  double recip=2./theta_sigma;
197  for(int i=0;i<size;i++) {
198  double d=dx[i];
199  double d1=dx[i+1];
200  stencil[i]=norm*(d*d*d*Gx[i]-d1*d1*d1*Gx[i+1]) - recip * DXS[i];
201  }
