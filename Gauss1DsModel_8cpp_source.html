210 void
211 Gauss1DsModel::pixel_hess_update(IdxT i, const Stencil &s, double dm_ratio_m1, double dmm_ratio, ParamT &grad, MatT &hess) const
212 {
213  /* Caclulate pixel derivative */
214  auto pgrad=make_param();
215  pixel_grad(i,s,pgrad);
216  double I = s.I();
217  /* Update grad */
218  grad += dm_ratio_m1*pgrad;
219  /* Update hess */
220  hess(0,0) += dm_ratio_m1 * I/s.sigma() * s.DXS(i);
221  hess(0,1) += dm_ratio_m1 * s.DX(i);
222  hess(0,3) += dm_ratio_m1 * I * s.DXSX(i);
223  hess(1,3) += dm_ratio_m1 * s.DXS(i);
224  hess(3,3) += dm_ratio_m1 * I * s.DXS2(i);
225  //This is the pixel-gradient dependent part of the hessian
226  for(IdxT c=0; c<hess.n_cols; c++) for(IdxT r=0; r<=c; r++)
227  hess(r,c) -= dmm_ratio * pgrad(r) * pgrad(c);
228 }

230 Gauss1DsModel::Stencil
231 Gauss1DsModel::initial_theta_estimate(const ImageT &im, const ParamT &theta_init) const
232 {
233  double x_pos=0, I=0, bg=0, sigma=0;
234  if (!theta_init.is_empty()) {
235  x_pos = theta_init(0);
236  I = theta_init(1);
237  bg = theta_init(2);
238  sigma = theta_init(3);
239  }
240  if(x_pos<=0 || x_pos>size) x_pos = im.index_max()+0.5; //Estimate position as the brightest pixel
241 
242  double min_sigma = lbound(3);
243  double max_sigma = ubound(3);
244  if(sigma<min_sigma || sigma>max_sigma){
245  //Pick an initial sigma in-between min and max for sigma
246  //This is a rough approximation
247  double eta=0.8; //Weight applies to min sigma in weighted average
248  sigma = eta*min_sigma + (1-eta)*max_sigma;
249  }
250  if(bg <= 0) bg = std::max(1.0, 0.75*im.min());
251  if(I <= 0) I = std::max(1.0, arma::sum(im) - bg*size);
252  return make_stencil(ParamT{x_pos, I, bg, sigma});
253 }
