191 void Gauss1DModel::pixel_hess_update(IdxT i, const Stencil &s, double dm_ratio_m1, double dmm_ratio, ParamT &grad, MatT &hess) const
192 {
193  auto pgrad = make_param();
194  pixel_grad(i,s,pgrad);
195  double I = s.I();
196  /* Update grad */
197  grad += dm_ratio_m1*pgrad;
198  /* Update hess */
199  hess(0,0) += dm_ratio_m1 * I/psf_sigma * s.DXS(i);
200  hess(0,1) += dm_ratio_m1 * pgrad(0) / I;
201  //This is the pixel-gradient dependent part of the hessian
202  for(IdxT c=0; c<hess.n_cols; c++) for(IdxT r=0; r<=c; r++)
203  hess(r,c) -= dmm_ratio * pgrad(r) * pgrad(c);
204 }

206 Gauss1DModel::Stencil
207 Gauss1DModel::initial_theta_estimate(const ImageT &im, const ParamT &theta_init) const
208 {
209  double x_pos = 0;
210  double I = 0;
211  double bg = 0;
212  if(theta_init.n_elem == num_params){
213  x_pos = theta_init(0);
214  I = theta_init(1);
215  bg = theta_init(2);
216  }
217 
218  if(x_pos <= 0 || x_pos > size || !std::isfinite(x_pos))
219  x_pos = im.index_max()+0.5;
220  if(bg <= 0 || !std::isfinite(bg))
221  bg = std::max(1.0, 0.75*im.min());
222  if(I <= 0 || !std::isfinite(I))
223  I = std::max(1.0, arma::sum(im) - bg*size);
224  return make_stencil(ParamT{x_pos, I, bg});
225 }
