102 bool cholesky(arma::mat &A)
103 {
104  //A comes in as upper triangular symmetric, but is converted to a lower triangular
105  //Cholesky decomposition format on output. This way we keep the original matrix around in
106  //the upper triangle while we write the decomposition into the lower triangle. Cool!
107  int size=static_cast<int>(A.n_rows);
108  arma::vec v(size-1);
109  for(int j=0;j<size;j++) {
110  //compute v: 0..j-1
111  for(int i=0;i<j;i++) v(i)=A(j,i)*A(i,i); //i<j so use upper triangular access
112  double dotp=0;
113  for(int i=0;i<j;i++) dotp+=A(j,i)*v(i);
114  A(j,j) = A(j,j)-dotp;
115  if (A(j,j)<=0) return false;
116  for(int i=j+1;i<size;i++) {
117  double dotp=0;
118  for(int k=0;k<j;k++) dotp += A(i,k)*v(k);

120  }
121  }
122  //zero out upper triangular
123  for(int j=1;j<size;j++) for(int i=0;i<j;i++) A(i,j) = 0; //i<j

128 bool modified_cholesky(arma::mat &A)
129 {
130  //A comes in as upper triangular symmetric, but is converted to a lower triangular
131  //Cholesky decomposition format on output. This way we keep the original matrix around in
132  //the upper triangle while we write the decomposition into the lower triangle. Cool!
133  //We are using the Gill Murray Wright 1981 method although a superior but more complex version
134  //by Schandel and Eskow 1999 exists too that we should eventually implement.
135  assert(A.is_square());
136  int size=static_cast<int>(A.n_rows);
137  arma::vec v(size);
138  double gamma = arma::max(arma::abs(A.diag()));
139  double xi = 0;
140  for(int j=1;j<size;j++) for(int i=0;i<j;i++) xi = std::max(xi, fabs(A(i,j))); //Maximum over off-diagonal elements of uppers symmetric matrix
141  double epsilon = std::numeric_limits<double>::epsilon();
142  double delta = epsilon*std::max(1.,gamma+xi);
143  double beta_sq = std::max(epsilon,std::max(gamma, xi / size));
144 // double beta = sqrt(beta_sq);
145  bool positive_definite=true;
146 // std::cout<<"Modified Cholesky Decomposition.\n";
147 // std::cout<<"Size: "<<size<<" Condition:"<<arma::cond(arma::symmatu(A))<<"Lambdas:"<<arma::eig_sym(arma::symmatu(A)).t()<<"\n";
148 // std::cout<<"gamma:"<<gamma<<" xi:"<<xi<<" epsilon:"<<epsilon<<" delta:"<<delta<<" beta^2:"<<beta_sq<<" beta:"<<beta<<"\n";
149 // std::cout<<"A:\n"<<A;
150  for(int j=0;j<size;j++) {
151  //compute v: 0..j-1
152  for(int i=0;i<j;i++) v(i)=A(j,i)*A(i,i); //i<j so use lower triangular access
153 // std::cout<<"v:"<<v.t()<<"\n";
154  double dotp=0;
155  for(int i=0;i<j;i++) dotp+=A(j,i)*v(i); // lower triangular access to pre-computed
156 // std::cout<<"A("<<j<<","<<j<<")="<<A(j,j)<<" dotp:"<<dotp<<std::endl;
157  v(j)=A(j,j)-dotp;
158  double theta = 0; //Theta will be the max value of the newly computed Cij column j.
159  for(int i=j+1;i<size;i++) {
160  double dotp=0;
161  for(int k=0;k<j;k++) dotp += A(i,k)*v(k);
162 // std::cout<<"i:"<<i<<" dotp:"<<dotp<<"\n";
163  double val = A(i,j)-dotp; //i>j so store to lower triangular
164  theta = std::max(theta, fabs(val));
165  A(i,j) = val;
166 // std::cout<<"A(i,j)=A("<<i<<","<<j<<")="<<val<<"\n";
167  }
168  //determine diagonal element
169  A(j,j) = std::max(fabs(v(j)), std::max(theta*theta/beta_sq, delta));
170 // if(j+1<=size-1) std::cout<<"c: "<<A(arma::span(j+1,size-1),j).t()<<"\n";
171  if (A(j,j)!=v(j)) {
172 // std::cout<<"v("<<j<<")="<<v(j)<<" theta:"<<theta<<" theta^2/beta^2:"<<theta*theta/beta_sq<<" delta:"<<delta<<"\n";
173 // std::cout<<"A("<<j<<","<<j<<")="<<A(j,j)<<"\n";
174  positive_definite=false;
175  v(j) = A(j,j);
176 // std::cout<<"Not positive definite!\n";
177  }
178  for(int i=j+1;i<size;i++) A(i,j)/=A(j,j); //Re-normalize by new diagonal element A(j,j)
179 // std::cout<<"A:\n"<<A;
180  }
181  //zero out upper triangular
182  for(int j=1;j<size;j++) for(int i=0;i<j;i++) A(i,j) = 0; //i<j
183  return positive_definite;
184 }
