180 Gauss2DsModel::pixel_grad(int i, int j, const Stencil &s, ParamT &pgrad) const
181 {
182  double I = s.I();
183  pgrad(0) = I * s.DX(i) * s.Y(j);
184  pgrad(1) = I * s.DY(j) * s.X(i);
185  pgrad(2) = s.X(i) * s.Y(j);
186  pgrad(3) = 1.;
187  pgrad(4) = I * (s.Y(j)*s.DXS(i) + s.X(i)*s.DYS(j));
188 }

191 void
192 Gauss2DsModel::pixel_grad2(int i, int j, const Stencil &s, ParamT &pgrad2) const
193 {
194  double I = s.I();
195  pgrad2(0) = I/s.sigmaX() * s.DXS(i) * s.Y(j);
196  pgrad2(1) = I/s.sigmaY() * s.DYS(j) * s.X(i);
197  pgrad2(2) = 0.;
198  pgrad2(3) = 0.;
199  pgrad2(4) = I * (s.X(i)*s.DYS2(j) + 2.*s.DXS(i)*s.DYS(j) + s.Y(j)*s.DXS2(i));
200 }

203 void
204 Gauss2DsModel::pixel_hess(int i, int j, const Stencil &s, MatT &hess) const
205 {
206  hess.zeros();
207  double I = s.I();
208  //On Diagonal
209  hess(0,0) = I/s.sigmaX() * s.DXS(i) * s.Y(j); //xx
210  hess(1,1) = I/s.sigmaY() * s.DYS(j) * s.X(i); //yy
211  hess(4,4) = I*(s.X(i)*s.DYS2(j) + 2.*s.DXS(i)*s.DYS(j) + s.Y(j)*s.DXS2(i)); //SS
212  //Off Diagonal
213  hess(0,1) = I * s.DX(i) * s.DY(j); //xy
214  hess(0,4) = I * (s.Y(j)*s.DXSX(i) + s.DX(i)*s.DYS(j)); //xS
215  hess(1,4) = I * (s.X(i)*s.DYSY(j) + s.DY(j)*s.DXS(i)); //yS
216  //Off Diagonal with respect to I
217  hess(0,2) = s.DX(i) * s.Y(j); //xI
218  hess(1,2) = s.DY(j) * s.X(i); //yI
219  hess(2,4) = s.Y(j)*s.DXS(i) + s.X(i)*s.DYS(j); //IS
220 }
