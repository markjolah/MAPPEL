52 VecT solve_profile_initial_step(const MatT &obsI, IdxT fixed_idx, double llh_delta)
53 {
54  auto Np = obsI.n_rows;
55  MatT free_hess(Np-1,Np-1);
56  VecT fixed_grad2(Np-1);
57  //obsI and free_hess are in symmetric upper-triangular format
58  for(IdxT j=0,jj=0; j<Np; j++) {
59  if(j!=fixed_idx) {
60  for(IdxT i=0,ii=0;i<=j;i++) {
61  if(i!=fixed_idx)
62  free_hess(ii++,jj) = -obsI(i,j);
63  }
64  jj++;
65  } else {
66  for(IdxT i=0,ii=0;i<Np;i++) {
67  if(i!=fixed_idx)
68  fixed_grad2(ii++) = (i<j) ? -obsI(i,j) : -obsI(j,i);
69  }
70  }
71  }
72 
73  VecT tangent = arma::solve(arma::symmatu(free_hess),-fixed_grad2);
74  double denom = obsI(fixed_idx,fixed_idx)+arma::dot(fixed_grad2,tangent);
75  double h = -sqrt(fabs(llh_delta/denom));
76 
77  VecT step(Np);
78  for(IdxT i=0,ii=0; i<Np; i++) step(i) = (i==fixed_idx) ? h : h*tangent(ii++);
79  return step;
80 }

189 VecT solve_TR_subproblem(const VecT &g, const MatT &H, double delta)
190 {
191  auto N = g.n_elem;
192  double g_norm = arma::norm(g);
193  MatT Hchol = H;
194  bool pos_def = cholesky(Hchol);
195 
196  if(pos_def) {
197  VecT newton_step = cholesky_solve(Hchol,-g); //this is s(0), i.e., lambda=0
198  if(arma::norm(newton_step)<=delta){
199  //[Case 1]: Full Newton Step
200  return newton_step;
201  } else {
202  //[Case 2]: Restricted Newton Step
203  //Attempt to arrive at bounds that avoids having to do an eigendecomposition in this case
204  double lambda_lb = 0;
205  double lambda_ub = g_norm/delta;
206  return solve_restricted_step_length_newton(g,H,delta,lambda_lb, lambda_ub);
207  }
208  } else {
209  //Indefinite hessian. Do eigendecomposition to better understand the hessian
210  VecT lambda_H;
211  MatT Q_H;
212  bool decomp_success = arma::eig_sym(lambda_H,Q_H, arma::symmatu(H)); //Compute eigendecomposition of symmetric matrix H
213  if(!decomp_success) throw NumericalError("Could not eigendecompose");
214  VecT g_hat = Q_H.t() * g; // g in coordinates of H's eigensystem
215  double delta2 = delta * delta; //delta^2
216  double lambda_min = lambda_H(0); //Minimum eigenvalue. lambda_H is guaranteed in decreasing order
217 
218  // compute multiplicity of lambda_min.
219  arma::uword Nmin = 1;
220  while(lambda_H(Nmin)==lambda_min) Nmin++;
221  //Compute ||P_min g'||^2
222  double g_min_norm = 0.;
223  for(arma::uword i=0; i<Nmin;i++) g_min_norm += g_hat(i)*g_hat(i);
224  g_min_norm = sqrt(g_min_norm);
225  if(g_min_norm>0) {
226  //[Case 3]: Indefinite hessian, general-position gradient
227  //The gradient extends into the lambda min subspace, so there is a pole at s(lambda_min)=inf
228  double lambda_lb = g_min_norm/delta - lambda_min;
229  double lambda_ub = g_norm/delta - lambda_min;
230  return solve_restricted_step_length_newton(g,H,delta,lambda_lb, lambda_ub);
231  } else {
232  //Compute s_perp_sq = ||P^perp_min s(lambda_min)||^2
233  VecT s_perp_lmin(N,arma::fill::zeros);
234  for(arma::uword i=Nmin; i<N; i++) s_perp_lmin += (g_hat(i)/(lambda_H(i)-lambda_min)) * Q_H.col(i);
235  double s_perp_lmin_normsq = arma::dot(s_perp_lmin,s_perp_lmin);
236  std::cout<<" s_perp_lmin_normsq: "<<s_perp_lmin_normsq<<"\n";
237  if(s_perp_lmin_normsq >= delta) {
238  //[Case 4]: Indefinite hessian, degenerate gradient, sufficient perpendicular step
239  double lambda_lb = -lambda_min;
240  double lambda_ub = g_norm/delta - lambda_min;
241  std::cout<<" {{Case 4: Indefinite hessian, degenerate gradient, sufficient perpendicular step}}";
242  std::cout<<" [lambda range: "<<lambda_lb<<"-"<<lambda_ub<<"]\n";
243  return solve_restricted_step_length_newton(g,H,delta,lambda_lb, lambda_ub);
244  } else {
245  //[Case 5]: Indefinite hessian, degenerate gradient, insufficient perpendicular step
246  //(i.e., The hard-hard case)
247  double tau = sqrt(delta2 - s_perp_lmin_normsq);
248  std::cout<<" {{Case 5: Indefinite hessian, degenerate gradient, insufficient perpendicular step}}";
249  std::cout<<" [tau: "<<tau<<" ] q1:"<<Q_H.col(0).t();
250  return s_perp_lmin + tau * Q_H.col(0); // s_perp(lambda_min) + tau * q_min; q_min is vector from S_min subspace
251  }
252  }
253  }

256 VecT solve_restricted_step_length_newton(const VecT &g, const MatT &H, double delta, double lambda_lb, double lambda_ub)
257 {
258  //Initially Hchol is the Cholesky decomposition at lambda=0
259  double lambda = clamp(std::max(1./delta, .5*(lambda_lb+lambda_ub)), lambda_lb, lambda_ub);
260  int max_iter = 50;
261 
262  for(int i=0; i<max_iter;i++) {
263  MatT R = H;
264  R.diag() += lambda;
265  bool is_pos = cholesky(R);
266  if(!is_pos) {
267  throw NumericalError("Bad Cholesky decomposition. Lambda is too small??.");
268  }
269  VecT p = cholesky_solve(R,-g);
270  MatT Rtri = R;
271  cholesky_convert_lower_triangular(Rtri);
272  VecT q = arma::solve(arma::trimatl(Rtri), p);
273  double norm_p = arma::norm(p);
274  double objective = 1/delta - 1/norm_p;
275 
276  if(std::fabs(objective) < tr_subproblem_newton_epsilon) return p;
277  double lambda_delta = (norm_p/arma::norm(q)) * (norm_p/arma::norm(q)) * ((norm_p-delta)/delta);
278  if(std::fabs(lambda_delta) < tr_subproblem_newton_epsilon) return p;
279  if(lambda==lambda_lb && lambda_delta<0) {
280  throw NumericalError("Bad lambda lower bound??");
281  }
282  if(lambda==lambda_ub && lambda_delta>0) {
283  throw NumericalError("Bad lambda upper bound??");
284  }
285  lambda += lambda_delta;
286  lambda = std::min(std::max(lambda,lambda_lb),lambda_ub);
287  }
288  throw NumericalError("Lambda search exceeded max_iter");
289 }
