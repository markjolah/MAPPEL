495 bool IterativeMaximizer<Model>::backtrack(MaximizerData &data)
496 {
497  double lambda = 1.0;
498  data.save_stencil(); //Save current theta in case we cannot improve
499  if (!data.step.is_finite()) {
500  std::ostringstream msg;
501  msg<<"Step is not finite: "<<data.step.t();
502  throw NumericalError(msg.str());
503  }
504  if( (data.has_fixed_parameters && arma::dot(data.grad%data.fixed_parameter_scalar, data.step)<=0) ||
505  arma::dot(data.grad, data.step)<=0){
506  //We are maximizing so we should be moving in direction of gradient not away
507  std::ostringstream msg;
508  msg<<"Backtrack with Negative Gradient. grad:"<<data.grad.t()<<" step:"<<data.step.t()<<" <grad, step>: "<<arma::dot(data.grad, data.step);
509  throw NumericalError(msg.str());
510  }
511  for(int n=0; n<max_backtracks; n++){
512  //Boundary conditions
513  auto new_theta = model.bounded_theta(model.reflected_theta(data.saved_theta() + lambda*data.step));
514  data.set_stencil(model.make_stencil(new_theta,false));
515  double can_rllh = methods::objective::rllh(model, data.im, data.stencil()); //candidate points log-lh
516 
517 // std::cout<<"\n [Backtrack:"<<n<<"]\n";
518 // std::cout.precision(15);
519 // data.saved_theta().t().raw_print(std::cout," Current Theta:");
520 // std::cout<<" Step: "<<data.step.t();
521 // std::cout<<" Lambda:"<<lambda<<"\n Scaled Step:"<<(lambda*data.step).t();
522 // std::cout<<" Proposed Theta: "<<new_theta.t();
523 // printf(" CurrentRLLH:%.9g ProposedRLLH:%.9g Delta(prop-cur):%.9g\n",data.rllh,can_rllh,can_rllh-data.rllh);
524 
525  bool in_bounds = model.theta_in_bounds(new_theta);
526  if(!in_bounds) {
527  std::ostringstream msg;
528  msg<<"New theta is not inbounds. old_theta:"<<data.saved_theta().t()<<" step:"<<data.step.t()
529  <<" new_theta:"<<new_theta.t();
530  throw NumericalError(msg.str());
531  }
532  if(!std::isfinite(can_rllh)) {
533  std::ostringstream msg;
534  msg<<"Candidate theta is inbounds but rllh is non-finite. new_theta:"<<new_theta.t()<<" rllh:"<<can_rllh;
535  throw NumericalError(msg.str());
536  }
537 
538  double old_lambda = lambda; //save old lambda
539  double linear_step = data.has_fixed_parameters ? lambda*arma::dot(data.grad%data.fixed_parameter_scalar, data.step) :
540  lambda*arma::dot(data.grad, data.step); //The amount we would go down if linear in grad
541  if (can_rllh >= data.rllh + alpha*linear_step) { //Must be a sufficient increase (convergence criteria)
542  //Success - Found a new point which is slightly better than where we were before, so update rllh
543  data.rllh = can_rllh; //We have not yet changed data.rllh that is still the old rllh
544  data.stencil().compute_derivatives(); //Now we commit to the expense of computing derivatives
545  data.record_iteration(); //Record a successful point
546  return false; //Tell caller to continue optimizing
547  } else {
548  data.record_backtrack(can_rllh); //Record a failed (backtracked) (unaccepted) point.
549  if (convergence_test(data)) {
550  //We converged at original iteration point. Unable to improve by backtracking on step direction.
551  //Original step was too large to converge, after backtracking it (or fun-val) became small enough to converge
552  data.restore_stencil(); //Restore back to original point
553  return true;
554  } else {
555  //Perform Backtrack using a quadratic approximation
556  double new_lambda = -.5*lambda*linear_step/(can_rllh - data.rllh - linear_step); //The minimum of a quad approx using cllh and linear_step
557  double rho = new_lambda/old_lambda; //Relative decrease
558  rho = std::min(std::max(rho,0.02),0.25); //Limit minimum and maximum relative decrease of rho
559  lambda = rho*old_lambda;
560  }
561  }
562  }
563  //Backtracking failed to converge in max_backtracks steps.
564  //But we otherwise did not meet convergence criteria.
565  data.restore_stencil();
566  data.record_exit(ExitCode::MaxBacktracks);
567  return true; //Tell caller to stop and return the original theta. Backtracking failed to improve it.
568 }
