226 template <class Model>
227 void estimate_posterior_stack(const Model &model, const ModelDataStackT<Model> &data_stack,
228  const ParamVecT<Model> &theta_init_stack, mcmc::MCMCDataStack &est)
229 {
230  auto Np = model.get_num_params();
231  est.Ndata = model.get_size_image_stack(data_stack);
232  est.initialize_arrays(Np);
233  auto Noversample = mcmc::num_oversample(est.Nsample,est.Nburnin,est.thin);
234  omp_exception_catcher::OMPExceptionCatcher catcher;
235  #pragma omp parallel
236  {
237  auto oversample = model.make_param_stack(Noversample);
238  VecT oversample_rllh(Noversample);
239  auto sample = model.make_param_stack(est.Nsample);
240  VecT sample_rllh(est.Nsample);
241  auto sample_mean = model.make_param();
242  auto sample_cov = model.make_param_mat();
243  auto credible_lb = model.make_param();
244  auto credible_ub = model.make_param();
245  #pragma omp for
246  for(IdxT n=0; n<est.Ndata; n++)
247  catcher.run([&]{
248  auto data = model.get_image_from_stack(data_stack,n);
249  mcmc::sample_posterior(model, data, model.initial_theta_estimate(data, theta_init_stack.col(n)), oversample, oversample_rllh);
250  mcmc::thin_sample(oversample, oversample_rllh, est.Nburnin, est.thin, sample, sample_rllh);
251  mcmc::estimate_sample_posterior(sample, sample_mean, sample_cov);
252  mcmc::compute_posterior_credible(sample, est.confidence, credible_lb, credible_ub);
253  est.credible_lb.col(n) = credible_lb;
254  est.credible_ub.col(n) = credible_ub;
255  est.sample.slice(n) = sample;
256  est.sample_rllh.col(n) = sample_rllh;
257  est.sample_mean.col(n) = sample_mean;
258  est.sample_cov.slice(n) = sample_cov;
259  });
260  }
261  catcher.rethrow();
262 }
