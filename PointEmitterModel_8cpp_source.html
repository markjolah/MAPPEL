275 PointEmitterModel::ParamT PointEmitterModel::reflected_theta(const ParamT &theta) const
276 {
277  check_param_shape(theta);
278  ParamT btheta = theta;
279  for(IdxT n=0;n<num_params;n++) {
280  if(std::isfinite(lbound(n))) {
281  if(std::isfinite(ubound(n))){//both valid bounds. Do reflection
282  double d = 2*(ubound(n)-lbound(n));
283  double w = std::fmod(std::fabs(theta(n)-lbound(n)), d);
284  btheta(n) = std::min(w,d-w)+lbound(n);
285  } else if (theta(n)<lbound(n)) {
286  btheta(n)=2*lbound(n)-theta(n); //valid lower bound only
287  }
288  } else if(theta(n)>ubound(n)) {
289  btheta(n)=2*ubound(n)-theta(n); //valid upper bound only
290  }
291  }
292  return btheta;
293 }
