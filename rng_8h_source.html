57 IdxT generate_poisson_large(RngT &rng, double mu)
58 {
59  UniformDistT uniform;
60  double c = 0.767 - 3.36/mu;
61  double beta = arma::datum::pi/sqrt(3.0*mu);
62  double alpha = beta*mu;
63  double k = log(c) - mu - log(beta);
64 
65  while(true) {
66  double u = 1-uniform(rng);//support: (0,1]
67  double x = (alpha - log((1.0 - u)/u))/beta;
68  double n = floor(x + 0.5);
69  if(n<0) continue;
70  double v = 1-uniform(rng);//support: (0,1]
71  double y = alpha - beta*x;
72  double temp = 1.0 + exp(y);
73  double lhs = y + log(v/(temp*temp));
74  double rhs = k + n*log(mu) - lgamma(n+1);
75  if (lhs <= rhs)
76  return n;
77  }
