177 void
178 Gauss2DsxyModel::pixel_grad(int i, int j, const Stencil &s, ParamT &pgrad) const
179 {
180  double I = s.I();
181  pgrad(0) = I * s.DX(i) * s.Y(j);
182  pgrad(1) = I * s.DY(j) * s.X(i);
183  pgrad(2) = s.X(i) * s.Y(j);
184  pgrad(3) = 1.;
185  pgrad(4) = I * (s.Y(j)*s.DXS(i) + s.X(i)*s.DYS(j));
186 }

189 void
190 Gauss2DsxyModel::pixel_grad2(int i, int j, const Stencil &s, ParamT &pgrad2) const
191 {
192  double I = s.I();
193  pgrad2(0) = I/s.sigmaX() * s.DXS(i) * s.Y(j);
194  pgrad2(1) = I/s.sigmaY() * s.DYS(j) * s.X(i);
195  pgrad2(2) = 0.;
196  pgrad2(3) = 0.;
197  pgrad2(4) = I * (s.X(i)*s.DYS2(j) + 2.*s.DXS(i)*s.DYS(j) + s.Y(j)*s.DXS2(i));
198 }

200 inline
201 void
202 Gauss2DsxyModel::pixel_hess(int i, int j, const Stencil &s, MatT &hess) const
203 {
204  hess.zeros();
205  double I = s.I();
206  //On Diagonal
207  hess(0,0) = I/s.sigmaX() * s.DXS(i) * s.Y(j); //xx
208  hess(1,1) = I/s.sigmaY() * s.DYS(j) * s.X(i); //yy
209  hess(4,4) = I*(s.X(i)*s.DYS2(j) + 2.*s.DXS(i)*s.DYS(j) + s.Y(j)*s.DXS2(i)); //SS
210  //Off Diagonal
211  hess(0,1) = I * s.DX(i) * s.DY(j); //xy
212  hess(0,4) = I * (s.Y(j)*s.DXSX(i) + s.DX(i)*s.DYS(j)); //xS
213  hess(1,4) = I * (s.X(i)*s.DYSY(j) + s.DY(j)*s.DXS(i)); //yS
214  //Off Diagonal with respect to I
215  hess(0,2) = s.DX(i) * s.Y(j); //xI
216  hess(1,2) = s.DY(j) * s.X(i); //yI
217  hess(2,4) = s.Y(j)*s.DXS(i) + s.X(i)*s.DYS(j); //IS
218 }
