62 Gauss2DModel::make_internal_1Dsum_estimator(IdxT dim, const ImageSizeT &size, const VecT &psf_sigma, const CompositeDist &prior)
63 {
64  std::type_index pos_dist = prior.component_types()[dim];
65  std::type_index beta_dist(typeid(prior_hessian::ScaledSymmetricBetaDist));
66  std::type_index normal_dist(typeid(prior_hessian::TruncatedNormalDist));
67  auto hyperparams = prior.params();
68  if(pos_dist == beta_dist){
69  double beta_pos = hyperparams(dim);
70  double mean_I = hyperparams(2);
71  double kappa_I = hyperparams(3);
72  double mean_bg = size((dim+1)%2) * hyperparams(4); //Amplify bg to account for summation
73  double kappa_bg = hyperparams(5);
74  return {size(dim),psf_sigma(dim),Gauss1DSumModelT::make_prior_beta_position(
75  size(dim), beta_pos, mean_I, kappa_I, mean_bg, kappa_bg)};
76  } else if(pos_dist == normal_dist) {
77  double sigma_pos = hyperparams(2*dim+1);
78  double mean_I = hyperparams(4);
79  double kappa_I = hyperparams(5);
80  double mean_bg = size((dim+1)%2) * hyperparams(6); //Amplify bg to account for summation
81  double kappa_bg = hyperparams(7);
82  return {size(dim),psf_sigma(dim),Gauss1DSumModelT::make_prior_normal_position(
83  size(dim), sigma_pos, mean_I, kappa_I, mean_bg, kappa_bg)};
84  } else {
85  std::ostringstream msg;
86  msg<<"Unknown position distribution: "<<pos_dist.name();
87  throw ParameterValueError(msg.str());
88  }

282 void Gauss2DModel::pixel_hess_update(int i, int j, const Stencil &s, double dm_ratio_m1, double dmm_ratio,
283  ParamT &grad, MatT &hess) const
284 {
285  /* Caclulate pixel derivative */
286  auto pgrad = make_param();
287  pixel_grad(i,j,s,pgrad);
288  double I = s.I();
289  /* Update grad */
290  grad += dm_ratio_m1*pgrad;
291  /* Update hess */
292  hess(0,0) += dm_ratio_m1 * I/psf_sigma(0) * s.DXS(i) * s.Y(j);
293  hess(0,1) += dm_ratio_m1 * I * s.DX(i) * s.DY(j);
294  hess(1,1) += dm_ratio_m1 * I/psf_sigma(1) * s.DYS(j) * s.X(i);
295  hess(0,2) += dm_ratio_m1 * pgrad(0) / I;
296  hess(1,2) += dm_ratio_m1 * pgrad(1) / I;
297  //This is the pixel-gradient dependent part of the hessian
298  for(IdxT c=0; c<hess.n_cols; c++) for(IdxT r=0; r<=c; r++)
299  hess(r,c) -= dmm_ratio * pgrad(r) * pgrad(c);
300 }

302 Gauss2DModel::Stencil
303 Gauss2DModel::initial_theta_estimate(const ImageT &im, const ParamT &theta_init,
304  const std::string &estimator_method)
305 {
306  double x_pos = 0;
307  double y_pos = 0;
308  double I = 0;
309  double bg = 0;
310  if(theta_init.n_elem == num_params){
311  if(theta_in_bounds(theta_init)) return make_stencil(theta_init);
312  x_pos = theta_init(0);
313  y_pos = theta_init(1);
314  I = theta_init(2);
315  bg = theta_init(3);
316  }
317  Gauss2DModel::ImageT x_im = arma::sum(im,0).t();
318  Gauss2DModel::ImageT y_im = arma::sum(im,1);
319  auto x_est = methods::estimate_max(x_model,x_im,estimator_method);
320  auto y_est = methods::estimate_max(y_model,y_im,estimator_method);
321 
322  if(x_pos <= lbound(0) || x_pos >= ubound(0) || !std::isfinite(x_pos)) x_pos = x_est.theta(0);
323  if(y_pos <= lbound(1) || y_pos >= ubound(1) || !std::isfinite(y_pos)) y_pos = y_est.theta(0);
324  if(I <= lbound(2) || I >= ubound(2) || !std::isfinite(I)) {
325  //max of X and Y est of I
326  I = std::max(x_est.theta(1), y_est.theta(1));
327  }
328  if(bg <= lbound(3) || bg >= ubound(3) || !std::isfinite(bg)) {
329  //mean of X and Y est of bg corrected for 1D vs 2D interpretation of bg
330  bg = .5*(x_est.theta(2)/size(1) + y_est.theta(2)/size(0));
331  }
332  return make_stencil(ParamT{x_pos, y_pos, I, bg});
333 }
