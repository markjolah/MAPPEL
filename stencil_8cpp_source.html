127 double gaussian_convolution(int x, int y, const MatT &data, const VecT &Xstencil,const VecT &Ystencil)
128 {
129  int size_x=data.n_cols;
130  int size_y=data.n_rows;
131  double gauss_val=0., sum=0.;
132  for(int i=0; i<size_x; i++) for(int j=0; j<size_y; j++) {
133  double Gxy=Xstencil(i-x+size_x-1)*Ystencil(j-y+size_y-1);
134  gauss_val+=Gxy*data(j,i);
135  sum+=Gxy;
136  }
137  gauss_val/=sqrt(sum); //this approximates the loss of information at the edges
138  return gauss_val;
